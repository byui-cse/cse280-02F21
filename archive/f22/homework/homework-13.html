tree = ['a', ['b', ['c', ['d'], ['e']], ['f', ['g'], ['h']]], ['i', ['j'], ['k', ['l'], ['m', ['n'], ['o']]]]]

# Print a single node
def process(v):
    print(v, end=" ")

def preorder(T):
    process(T[0])
    for child in T[1:]:
        preorder(child)

preorder(tree) # should result in: a b c d e f g h i j k l m n o

# Recognized bitstrings whose third-to-last bit is a 0
machine = '''
digraph {
    layout=dot;
    rankdir=LR;
    ranksep=1;
    size=6;
    start [shape=rect, label="Start", color="red", fontcolor="red", fontsize=12, width=0.5, height=0.25];
    subgraph {
        node [shape=circle]; s 0 00 01;
    }
    subgraph {
        rank=sink;
        node [shape=doublecircle]; 000 001 010 011;
    }
    start->s [color=red];
    s->s [label=1];
    s->0 [label=0];
    0->00 [label=0];
    0->01 [label=1];
    00->000 [label=0];
    00->001 [label=1];
    01->010 [label=0];
    01->011 [label=1];
    000->000 [label=0];
    000->001 [label=1];
    001->010 [label=0];
    001->011 [label=1, labelfloat="true"];
    010->00 [label=0];
    010->01 [label=1];
    011->0 [label=0];
    011->s [label=1];
}
'''
from graphviz import Source
Source(machine)
